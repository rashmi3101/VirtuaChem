 66 	/**
 67 	 * Substitutes all occurrences of '{n}' in the given string by the n-th
 68 	 * argument.
 69 	 * @private
 70 	 * @param {String} str String to be substituted.
 71 	 * @param {Any} [arguments]
 72 	 * @returns {String}
 73 	 */
 74 	function subst(str) {
 75 		$.each(Array.prototype.slice.call(arguments, 1), function(i, v) {
 76 			str = str.replace(new RegExp('\\{' + i + '\\}', 'g'), v);
 77 		});
 78 		return str;
 79 	}

137 	/**
138 	 * Returns the next item in the given ring, relative to the given item.
139 	 * Returns an empty jQuery object if the given item is not within the ring.
140 	 * @private
141 	 * @param {jqObject} $item
142 	 * @param {jqObject} $itemRing
143 	 * @returns {jqObject}
144 	 */
145 	Toolbar.$nextItem = function($item, $itemRing) {
146 		var i;
147 		if ($itemRing) {
148 			i = $itemRing.index($item) + 1;
149 			if (i > 0) {
150 				return $($itemRing[i % $itemRing.length]);
151 			}			
152 		}
153 		return $();
154 	};

172 		}

345 	/**
346 	 * Stops the current menu effect.
347 	 * @private
348 	 * @param {Object} menu
349 	 * @param {Object} [opt]
350 	 * @param {Boolean} [opt.clearQueue=false]
351 	 *        Whether to clear queued animations for the menu or not.
352 	 * @param {Boolean} [opt.jumpToEnd=false]
353 	 *        Whether to complete the current animation immediately or not.
354 	 * @returns {Function} Chainable.
355 	 */
356 	Toolbar.stopMenuFx = function(menu, opt) {
357 		if (menu.isShowFxInProgress || menu.isHideFxInProgress) {
358 			opt = opt || {};
359 			opt.clearQueue = Boolean(opt.clearQueue);
360 			opt.jumpToEnd = Boolean(opt.jumpToEnd);
361 			menu.$body.stop(opt.clearQueue, opt.jumpToEnd);
362 			if (menu.curFx === 'blind' || menu.curFx === 'fold') {
363 				Toolbar.stopFxWrapper(menu, opt.clearQueue, opt.jumpToEnd);
364 			} else if (menu.curFx === 'explode') {
365 				Toolbar.clearExplodeFxCompound();
366 			}
367 			menu.isShowFxInProgress = false;
368 			menu.isHideFxInProgress = false;
369 			menu.curFx = undefined;
370 		}
371 		return Toolbar;
372 	};

430 	/**
431 	 * Gets or sets this toolbar's options. Call with an empty object to restore
432 	 * default options. Expects an object with toolbar options, general effect
433 	 * options and options specific to an effect. Visit the jQueryUI API doc
434 	 * for further info.
435 	 * 
436 	 * @see <a href="http://api.jqueryui.com/category/effects/">jq UI FX API</a>
437 	 * 
438 	 * @private
439 	 * @function
440 	 * 
441 	 * @param {Object} [options]
442 	 * 
443 	 * @param {Boolean} [options.isRounded]
444 	 *        Whether this toolbar has rounded ends.
445 	 *        
446 	 * @param {Boolean} [options.isAboveMenus]
447 	 *        If true, menus are docked below the toolbar, otherwise menus are
448 	 *        docked above the toolbar.
449 	 *        
450 	 * @param {Boolean} [options.isSkippable]
451 	 *        Provides keyboard and screen-reader users an extra link allowing
452 	 *        them to skip toolbar navigation. Set 'skipNavigationTarget' to
453 	 *        make this feature even more useful.
454 	 *        
455 	 * @param {jqObject} [options.skipNavigationTarget]
456 	 *        Id or name of the element focused when keyboard or screen-reader
457 	 *        users skip the toolbar links. Per default an empty span succeeding
458 	 *        the toolbar's underlying list element. 
459 	 * 
460 	 * @param {Object} [options.showMenuFx]
461 	 *        jQueryUI effect specific options for showing menus.
462 	 * 
463 	 * @param {String} [options.showMenuFx.effect]
464 	 *        Name of the jQueryUI effect used to show menus.
465 	 * 
466 	 * @param {Number} [options.showMenuFx.duration]
467 	 *        Duration of the jQueryUI effect used to show a menu.
468 	 * 
469 	 * @param {String} [options.showMenuFx.easing]
470 	 *        jQueryUI effect easing used to show a menu.
471 	 * 
472 	 * @param {Object} [options.hideMenuFx]
473 	 *        jQueryUI effect specific options for hiding menus.
474 	 * 
475 	 * @param {String} [options.hideMenuFx.effect]
476 	 *        Name of the jQueryUI effect used to hide menus.
477 	 * 
478 	 * @param {Number} [options.hideMenuFx.duration]
479 	 *        Duration of the jQueryUI effect used to hide a menu.
480 	 * 
481 	 * @param {String} [options.hideMenuFx.easing]
482 	 *        jQueryUI effect easing used to hide a menu.
483 	 * 
484 	 * @returns {Toolbar|Object} This toolbar or it's options.
485 	 */
486 	Toolbar.prototype.options = (function() {
487 		function applyDirectionDefaults(fxOptions, isToolbarAboveMenus) {
488 			if (Toolbar.isOrientedFx(fxOptions.effect)) {
489 				fxOptions.direction = fxOptions.direction || 'horizontal';
490 			} else if (Toolbar.isVectoredFx(fxOptions.effect)) {
491 				fxOptions.direction = fxOptions.direction ||
492 					(isToolbarAboveMenus ? 'up' : 'down');
493 			}
494 		}
495 		return function(options) {
496 			if (!options) { return this.opt; }
497 			this.opt = Toolbar.defaultOptions();
498 			$.extend(true, this.opt, options);
499 			applyDirectionDefaults(this.opt.showMenuFx, this.opt.isAboveMenus);
500 			applyDirectionDefaults(this.opt.hideMenuFx, this.opt.isAboveMenus);
501 			return this.applyOptions();
502 		};
503 	}());

561 	/**
562 	 * Transforms the given elements into items (toolbar or menu items).
563 	 * @private
564 	 * @returns {Toolbar} Chainable.
565 	 */
566 	Toolbar.prototype.createItems = function() {
567 		var $items = this.$items();
568 		$items.addClass(ITEM_CLASS);
569 		$items.each($.proxy(function(i, el) {
570 			this.observeItem($(el));
571 		}, this));
572 		this.$toolbarItems({ excludeMenuLeadItems: true })
573 			.each($.proxy(function(i, el) {
574 				this.observeToolbarItem($(el));
575 			}, this));
576 		return this;
577 	};

579 	/**
580 	 * Transforms the given elements into menus.
581 	 * @private
582 	 * @returns {Toolbar} Chainable.
583 	 */
584 	Toolbar.prototype.createMenus = function() {
585 		var $menus = this.$menus();
586 		$menus.addClass(MENU_CLASS);
587 		$menus.each($.proxy(function(i, el) {
588 			el = $(el);
589 			this.menus.push({
590 				$body: el,
591 				$lead: el.prev(),
592 				dockRight: el.hasClass(DOCK_MENU_RIGHT_CLASS)				
593 			});
594 			this.observeMenu(this.menus[this.menus.length - 1]);
595 		}, this));
596 		return this;
597 	};

621 	/**
622 	 * Docks the given menu body to the corresponding lead item.
623 	 * @private
624 	 * @param {Object} menu
625 	 * @returns {Toolbar} Chainable.
626 	 */
627 	Toolbar.prototype.placeMenu = function(menu) {
628 		var left = menu.dockRight ?
629 			menu.$lead.outerWidth() - menu.$body.outerWidth() :
630 			0;
631 		if (this.opt.isRounded) {
632 			if (this.isLeftEndMenu(menu)) {
633 				left += menu.$lead.outerHeight() / 2;
634 			} else if (this.isRightEndMenu(menu)) {
635 				left -= menu.$lead.outerHeight() / 2;
636 			}			
637 		}
638 		menu.$body.css('left', left + 'px');
639 		menu.$body.css('top', this.opt.isAboveMenus ?
640 			menu.$lead.outerHeight() + 'px' :
641 			-menu.$body.outerHeight() + 'px');
642 		return this;
643 	};

645 	/**
646 	 * Resets the given menu to it's initial state (before any effects had
647 	 * been applied to the menu).
648 	 * @private
649 	 * @param {Object} menu
650 	 * @returns {Toolbar} Chainable.
651 	 */
652 	Toolbar.prototype.resetMenu = function(menu) {
653 		menu.$body.removeAttr('style');
654 		if (menu.$body.parent().hasClass(JQ_UI_FX_WRAPPER_CLASS)) {
655 			Toolbar.removeFxWrapper(menu);
656 		}
657 		this.placeMenu(menu);
658 		return this;
659 	};

661 	/**
662 	 * Shows the given menu using the specified effect. Resolves when the effect
663 	 * has been completely rendered.
664 	 * @private
665 	 * @param {Object} menu Menu to be shown.
666 	 * @param {Object} fxOptions Effect options.
667 	 * @returns {jqPromise} Deferrable.
668 	 */
669 	Toolbar.prototype.showMenu = function(menu, fxOptions) {
670 		var $d = $.Deferred(),
671 			/** @ignore */
672 			fxDone = function() {

844 						excludeDisabled: true

846 				break;

930 	/**
931 	 * Adds mouse event observers to the given menu.
932 	 * @private
933 	 * @param {Object} menu
934 	 * @returns {Toolbar} Chainable.
935 	 */
936 	Toolbar.prototype.observeMenuMouseEvents = function(menu) {
937 		//Some of the jQueryUi effects are a bit tricky to apply to the menu
938 		//because they animate the menu body in a way where the lead item
939 		//looses focus even if the mouse does not move outside that item,
940 		//e.g. when the menu body is moved in parts from outside in, layered
941 		//above the lead item (explode-effect).
942 		//This can cause infinite animation loops. To protect the user from
943 		//that undesired behavior, events are going to be ignored as long as
944 		//the effect is being rendered.
945 		//This on the other hand causes open menus not being closed when the
946 		//user moves the mouse pointer from one lead item to another one
947 		//before the first lead item's menu effect has finished. So when a
948 		//lead item is hovered, all menus not belonging to the hovered item
949 		//are explicitly closed if they are visible. In additon, all menus are
950 		//closed when hovering a normal toolbar item or hitting the escape key
951 		//(either locally, when an item is hovered, or globally).
952 		menu.$lead.on('mouseenter', $.proxy(function(ev) {
953 			//do not hide the menu when mouse moves from menu body to lead
954 			if (menu.hideMenuTimer) {
955 				root.clearTimeout(menu.hideMenuTimer);
956 				menu.hideMenuTimer = undefined;
957 			} else if (!(menu.isShowFxInProgress &&
958 					Toolbar.isFocusHijackingFx(this.opt.showMenuFx.effect))) {
959 				this.menuLeadItemFocusedIn(menu);
960 			}
961 		}, this));
962 		menu.$lead.on('mouseleave', $.proxy(function(ev) {
963 			if (!(menu.isShowFxInProgress &&
964 					Toolbar.isFocusHijackingFx(this.opt.showMenuFx.effect))) {
965 				menu.hideMenuTimer = root.setTimeout(
966 					$.proxy(function() {
967 						menu.hideMenuTimer = undefined;
968 						this.menuLeadItemFocusedOut(menu);
969 					}, this),
970 					HIDE_MENU_DELAY);
971 			}
972 		}, this));
973 		menu.$body.on('mouseenter', function(ev) {
974 			//do not hide the menu when mouse moves from lead to menu body
975 			root.clearTimeout(menu.hideMenuTimer);
976 		});
977 		menu.$body.on('mouseleave', $.proxy(function(ev) {
978 			menu.hideMenuTimer = root.setTimeout(
979 				$.proxy(function() {
980 					menu.hideMenuTimer = undefined;
981 					this.menuFocusedOut(menu);
982 				}, this),
983 				HIDE_MENU_DELAY);
984 		}, this));
985 		return this;
986 	};

1006 			}

1009 		return this;
